        $(document).ready(function() {

            var videos = [

                { type: 'video/mp4', src: 'https://freight.cargo.site/m/Q1918396806189591901251071833578/San-Joaquin-kit-fox.mp4' },

                { type: 'video/mp4', src: 'https://freight.cargo.site/m/E1918396806226485389398490936810/san-bruno-elfin-butterfly.mp4' },

                { type: 'video/mp4', src: 'https://freight.cargo.site/m/W1918396806244932133472200488426/Santa-Barbara-Song-Sparrow.mp4' },

                { type: 'video/mp4', src: 'https://freight.cargo.site/m/J1918396806263378877545910040042/thicktail-chub.mp4' }

            ];


            function getRandomVideo() {

                return videos[Math.floor(Math.random() * videos.length)];

            }


            function setVideoSource(videoElement, src, type) {

                var source = document.createElement('source');

                source.src = src;

                source.type = type;

                videoElement.empty(); // Clear existing sources

                videoElement.append(source);

            }


            function playRandomVideo() {

                var video = $('#myVideo')[0]; // Get the video DOM element

                var randomVideo = getRandomVideo();

                setVideoSource($(video), randomVideo.src, randomVideo.type);

                video.play().catch(function(error) {

                    console.log('Playback failed:', error);

                });

            }


            // Initial video load

            playRandomVideo();


            // Event listener to load a new video when the current one ends

            $('#myVideo').on('ended', function() {

                playRandomVideo();

            });

        });

    </script>
